\subsection{Advection-Diffusion-Reaction Equations}
\aframe{
\begin{itemize}
  \item Partition the interval $[0,T]$ as $0=t_0<t_1<\ldots<t_N=T$ uniformly
        with a fixed $\Delta t$. \pause
  \item $u_i^n$ denotes the $i$-th concentration at time $t_n$,
        $n=0,1,\ldots, N$. \pause
  \item The partial derivative w.r.t. $t$ is approximated as
        \[
        \dfrac{\partial u_i^n}{\partial t} \approx \dfrac{u^{n+1}_i-u_i^{n}}{\Delta t} \quad\text{(Backward
        Euler Approx.)}
        \] \pause
\end{itemize}
%
The weak formulation is
\begin{align*}
  &\int_\Omega\dfrac{1}{\Delta t}\sum_{i=1}^3\left[(u_i^{n+1}-u_i^n)v_i
    +w\cdot\nabla u_i^{n+1}v_i+\epsilon\nabla u_i^{n+1}\cdot\nabla v_i\right]dx\\
  +&\int_\Omega K\left(u_1^{n+1}u_2^{n+1}v_1+u_1^{n+1}u_2^{n+1}v_2-
     u_1^{n+1}u_2^{n+1}v_3+u_3^{n+1}v_3\right)dx\\
  -&\int_\Omega\sum_{i=1}^3f_iv_idx=0
\end{align*}
}
